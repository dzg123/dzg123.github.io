            getMethodName(),
            Arrays.asList(
                annotatedDataParam("data"),

            ),

    }


    @Override
    public MethodSpec buildSqlProviderMethod() {
        return providerMethod(getMethodName(), CodeBlock.builder()
            .add("return SCRIPT(")
            .add("\n  UPDATE(TABLE_NAME),")
            .add("\n  SET(")
            .add(resolveUpdateSetterCodeBlock())
            .add("\n  ),")
            .add("\n  WHERE_CLAUSE_BY_PARAM")
            .add("\n)")
            .build());
    }


    protected CodeBlock resolveUpdateSetterCodeBlock() {
        CodeBlock.Builder setters = CodeBlock.builder();
        iterate(
            tableContext.getColumns().stream().filter(this::shouldUpdate).iterator(),
            (col, idx) -> {
                setters.add("\n    $S",
                    format("%s = %s,", col.getColumnIdentifier(), valuePlaceholder(col, "data")));
            },
            (hasNext) -> {
                if (hasNext) {
                    setters.add(",");
                }
            }
        );

public class UpdateByUniqueKey extends UpdateFeatureBase implements GroupFeature {


    protected final UniqueKeyGroupContext uniqueKeyGroupContext;


    public UpdateByUniqueKey(TableDefinition ctx) {
        super(ctx);
        this.uniqueKeyGroupContext = new UniqueKeyGroupContext(ctx);
    }


    @Override
    public boolean isSupported() {
        return tableContext.isUpdatable()
            && tableContext.hasUniqueKey();
    }


    @Override
    public MethodSpec[] buildMapperMethod() {
        return uniqueKeyGroupContext.getUniqueKeyGroups()
            .keySet()
            .stream()
            .map(this::buildMapperMethod)
            .toArray(MethodSpec[]::new);
    }


    @Override
    public MethodSpec[] buildSqlProviderMethod() {
        return uniqueKeyGroupContext.getUniqueKeyGroups()
            .keySet()
            .stream()
            .map(this::buildSqlProviderMethod)
            .toArray(MethodSpec[]::new);
    }


    protected boolean shouldUpdate(TableColumnDefinition col) {
        return col.isUpdatable()
            && !col.isPrimaryKey()
            && !col.isShardingKey()
            && !col.isUniqueKey()
            && !col.isLargeObject();
    }


    protected String getMethodName(String groupKey) {
        return "updateBy" + groupKey;
    }


    protected MethodSpec buildMapperMethod(String groupKey) {
        return mapperMethod(TypeName.INT,
            getMethodName(groupKey),
            dataParam("data"),
            sqlProviderAnnotation(UpdateProvider.class));
    }


    protected MethodSpec buildSqlProviderMethod(String groupKey) {
        return providerMethod(
            getMethodName(groupKey),
            resolveSqlProviderCode(groupKey));
    }


    protected CodeBlock resolveSqlProviderCode(String groupKey) {
        CodeBlock.Builder setters = CodeBlock.builder();
        iterate(
            tableContext.getColumns().stream().filter(this::shouldUpdate).iterator(),
            (col, idx) -> {
                setters.add("\n    $S",
                    format("%s = %s,", col.getColumnIdentifier(), valuePlaceholder(col)));
            },
            (hasNext) -> {
                if (hasNext) {
                    setters.add(",");
                }
            }
        );
        return script(builder -> {
            builder
                .add("\n  UPDATE(TABLE_NAME),")
                .add("\n  SET(")
                .add(setters.build())
                .add("\n  ),")
                .add("\n  WHERE($S)", whereClauseByUniqueKey(groupKey));
        });
    }


    protected String whereClauseByUniqueKey(String groupKey) {
        return uniqueKeyGroupContext.getUniqueKeyGroups().get(groupKey)
            .stream()
            .map(col -> String.format("%s = #{%s}",
                col.getColumnIdentifier(),
                col.getPropertyName()))
            .collect(Collectors.joining(" AND "));
    }


}

    public VersionUpdateByPrimaryKey(TableDefinition ctx) {
        super(ctx);
    }


    @Override
    public boolean isSupported() {
        return tableContext.isUpdatable()

            && tableContext.hasVersionColumn();
    }


    @Override
    public MethodSpec buildMapperMethod() {
        return mapperMethod(TypeName.INT,
            getMethodName(),
            dataParam("data"),
            sqlProviderAnnotation(UpdateProvider.class));
    }


    @Override
    public MethodSpec buildSqlProviderMethod() {


        CodeBlock.Builder setters = CodeBlock.builder();
        iterate(
            tableContext.getColumns().stream().filter(this::shouldUpdate).iterator(),
            (col, idx) -> {
                if (col.isVersionColumn()) {
                    setters.add("\n    $S",
                        format("%s = %s + 1,", col.getColumnIdentifier(), col.getColumnIdentifier()));
                } else {
                    setters.add("\n    $S",
                        format("%s = %s,", col.getColumnIdentifier(), valuePlaceholder(col)));
                }
            },
            (hasNext) -> {
                if (hasNext) {
                    setters.add(",");
                }
            }
        );


        return providerMethod(getMethodName(), CodeBlock.builder()
            .add("return SCRIPT(")
            .add("\n  UPDATE(TABLE_NAME),")
            .add("\n  SET(")
            .add(setters.build())
            .add("\n  ),")
            .add("\n  WHERE($S)", resolveWhereClause())
            .add("\n)")
            .build());
    }


    protected boolean shouldUpdate(TableColumnDefinition col) {
        return col.isUpdatable()
            && !col.isPrimaryKey()
            && !col.isShardingKey()
            && !col.isLargeObject();
    }


    protected String resolveWhereClause() {


        TableColumnDefinition sk = tableContext.findShardingKeyColumn().orElse(null);


        TableColumnDefinition pk = tableContext.findPrimaryKeyColumn()
            .orElseThrow(() -> new RuntimeException("Primary key not found in context."));


        TableColumnDefinition vr = tableContext.findVersionColumn()
            .orElseThrow(() -> new RuntimeException("Version column not found in context."));


        if (sk != null) {
            return String.format("%s = #{%s} AND %s = #{%s} AND %s = #{%s}",
                sk.getColumnIdentifier(), sk.getPropertyName(),
                pk.getColumnIdentifier(), pk.getPropertyName(),
                vr.getColumnIdentifier(), vr.getPropertyName());
        } else {
            return String.format("%s = #{%s} AND %s = #{%s}",
                pk.getColumnIdentifier(), pk.getPropertyName(),
                vr.getColumnIdentifier(), vr.getPropertyName());
        }
    }


}

@SuppressWarnings("rawtypes")
public abstract class BaseParam<TParam extends BaseParam, TCriteria extends BaseCriteria> {


    @Getter
    protected boolean paged;


    @Getter
    protected long pageNo;


    @Getter
    protected long pageSize;


    @Getter
    protected long offset;


    @Getter
    protected long limit;


    @Getter
    protected List<TCriteria> orCriteria = new LinkedList<>();


    @Getter
    protected SqlExpr whereClause;


    @Getter
    protected List<SqlColumn> distinctCriteria = new LinkedList<>();


    @Getter
    protected List<Criterion> orderByCriteria = new LinkedList<>();


    protected abstract TParam getSelf();


    protected abstract TCriteria createCriteriaInternal();


    public TCriteria createCriteria() {
        TCriteria criteria = createCriteriaInternal();
        if (this.orCriteria.size() == 0) {
            this.orCriteria.add(criteria);
        }
        return criteria;
    }


    public TCriteria or() {
        TCriteria criteria = this.createCriteriaInternal();
        this.orCriteria.add(criteria);
        return criteria;
    }


    public void or(TCriteria criteria) {
        this.orCriteria.add(criteria);
    }


    public TParam clear() {
        this.paged = false;
        this.pageNo = 0;
        this.pageSize = 0;
        this.offset = 0;
        this.limit = 0;
        this.orCriteria.clear();
        this.distinctCriteria.clear();
        this.orderByCriteria.clear();
        this.whereClause = null;
        return getSelf();
    }


    public TParam distinct(SqlColumn... columns) {
        for (SqlColumn col : columns) {
            Objects.requireNonNull(col, "Require column not null.");
            this.distinctCriteria.add(col);
        }
        return getSelf();
    }


    public TParam where(SqlExpr whereClause) {
        this.whereClause = whereClause;
        return getSelf();
    }


    public TParam orderBy(SqlColumn column, SortType sort) {
        Objects.requireNonNull(column, "Require column not null.");
        Objects.requireNonNull(sort, "Require sort not null.");
        this.orderByCriteria.add((Criterion) column.orderBy(sort));
        return getSelf();
    }


    public TParam page(long pageNo, long pageSize) {
        this.paged = true;
        this.pageNo = pageNo < 1 ? 1 : pageNo;
        this.pageSize = pageSize < 1 ? 1 : pageSize;
        this.offset = this.pageSize * (this.pageNo - 1);
        this.limit = this.pageSize;
        return getSelf();
    }


    public TParam pageNo(long pageNo) {
        return this.page(pageNo, this.pageSize);
    }


    public TParam pageSize(long pageSize) {
        return this.page(this.pageNo, pageSize);
    }


}

public abstract class BaseSqlProvider {


    protected final static String WHERE_CLAUSE_COMBINE =
        CHOOSE(
            WHEN("param.whereClause != null",
                TRIM(attrs(prefixOverrides("AND|OR"), suffixOverrides("AND|OR")),
                    QUERY_EXPR("param.whereClause")
                )
            ),
            OTHERWISE(
                FOREACH(attrs(collection("param.orCriteria"), item("orCriteria"), separator(" OR ")),
                    IF("orCriteria.valid",
                        TRIM(attrs(prefix("("), suffix(")"), prefixOverrides("AND")),
                            FOREACH(
                                attrs(collection("orCriteria.criteria"), item("criterion"), separator(" AND ")),
                                QUERY_EXPR("criterion")
                            )
                        )
                    )
                )
            )
        );




    protected final static String WHERE_CLAUSE_BY_PARAM =
        WHERE(WHERE_CLAUSE_COMBINE);


    protected final static String WHERE_CLAUSE_EXPR =
        WHERE(QUERY_EXPR("queryExpr"));


    protected final static String ORDER_BY_CLAUSE_BY_PARAM =
        IF("param.orderByCriteria != null and param.orderByCriteria.size() > 0",
            ORDER_BY,
            FOREACH(
                attrs(collection("param.orderByCriteria"), item("criterion"), separator(",")),
                QUERY_EXPR("criterion"))
        );


    protected final static String LIMIT_CLAUSE_BY_PARAM =
        IF("param.paged", "LIMIT #{param.limit} OFFSET #{param.offset}");


    protected final static String LIMIT_ONE =
        LIMIT(1);


    protected static String WHERE_CLAUSE_BY_PARAM_AND_EXCLUDE_DELETED(String columnName, int deleteStatus) {
        return WHERE(
            CHOOSE(
                WHEN("param.whereClause != null",
                    TRIM(prefixOverrides("() AND"),
                        ENCLOSE(QUERY_EXPR("param.whereClause")),
                        AND_EXCLUDE_DELETED(columnName, deleteStatus)
                    )
                ),
                WHEN("param.orCriteria != null and param.orCriteria.size() > 0",
                    TRIM(prefixOverrides("() AND"),
                        ENCLOSE(
                            TRIM("",
                                FOREACH(attrs(collection("param.orCriteria"), item("orCriteria"), separator(" OR ")),
                                    IF("orCriteria != null and orCriteria.isValid()",
                                        TRIM(attrs(prefix("("), suffix(")"), prefixOverrides("AND")),
                                            FOREACH(
                                                attrs(collection("orCriteria.criteria"), item("criterion"), separator(" AND ")),
                                                QUERY_EXPR("criterion")
                                            )
                                        )
                                    )
                                )
                            )
                        ),
                        AND_EXCLUDE_DELETED(columnName, deleteStatus)
                    )
                ),
                OTHERWISE(
                    TRIM(prefixOverrides("AND"),
                        AND_EXCLUDE_DELETED(columnName, deleteStatus)
                    )
                )
            )
        );
    }


    protected static String COLUMNS_CLAUSE_BY_PARAM(String... columns) {
        return CHOOSE(
            WHEN("param.distinctCriteria != null and param.distinctCriteria.size() > 0",
                DISTINCT,
                FOREACH(
                    attrs(collection("param.distinctCriteria"), item("column"), separator(",")),
                    "${column.name}"
                )
            ),
            OTHERWISE(
                COLUMNS(columns)
            )
        );
    }


    protected static String AND_EXCLUDE_DELETED(String columnName, int deleteStatus) {
        return FORMAT("AND %s != %d", columnName, deleteStatus);
    }


    protected static String ENCLOSE(String inner) {
        return FORMAT("(%s)", inner);
    }


    protected static String QUERY_EXPR(String exprArg) {
        return FOREACH(attrs(collection(exprArg), item("expr")),
            CHOOSE(
                WHEN("expr.isLiteralExpr()", "${expr.literal}"),
                WHEN("expr.isIdentifierExpr()", "${expr.name}"),
                WHEN("expr.isValueExpr()",
                    CHOOSE(
                        WHEN("expr.isSingleValue", "#{expr.value}"),
                        WHEN("expr.isVector",
                            FOREACH(
                                attrs(collection("expr.value"), item("expValue"), open("("), close(")"), separator(",")),
                                "#{expValue}")
                        )
                    )
                )
            )
        );
    }
